### vue 基础使用

```html
<body>
  <!-- mvvm view -->
  <div id="app">{{msg}}</div>
  <script src="./node_modules/vue/dist/vue.js"></script>
  <script>
    // vm相当于viewmodel
    let vm = new Vue({
      el: "#app", // 告诉vm实例作用哪个dom元素
      // model  定义数据节点
      data() {
        return {
          msg: "hello vue",
        };
      },
    });
    vm.msg = "hahah"; // 数据驱动视图 数据改变 视图自动更新
  </script>
</body>
```
