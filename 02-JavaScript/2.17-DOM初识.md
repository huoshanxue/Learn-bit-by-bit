2、事件-----行为

三要素：事件类型(具体什么行为)、事件源（谁身上发生该行为）、事件处理

例如：打开电脑 电脑 打开电脑按钮 电脑启动

3、 js 程序如何处理事件
事件源.事件 = 事件处理函数

事件示例：

#### 点击事件 onclick

```HTML
<input type="button" value="点我" id="btn"/>
  <script>
    //单击按钮 弹窗hello world
    //事件源.事件 = 事件处理函数
    // 找事件源元素
    var btn = document.getElementById('btn');
    // 绑定事件
    btn.onclick = function() {
        // 事件发生后执行的逻辑
        alert('hello world');
    }
  </script>
```

#### 加载事件 onload

js 写在 html 之前，需要用到 onload 事件，待页面加载完成之后才执行。

```html
<script>
  // 页面加载
  window.onload = function() {
    // 单击按钮 弹窗hello world
    //事件源.事件 = 事件处理函数
    // 找事件源元素
    var btn = document.getElementById('btn');
    // 绑定事件
    btn.onclick = function() {
        // 事件发生后执行的逻辑
        alert('hello world');
    }
  }
</script>
</head>

<body>
    <input type="button" value="点我" id="btn" />
</body>
```

#### 鼠标移入/移出事件 onmouseover/onmouseout

```html
<style>
  * {
    padding: 0;
    margin: 0;
  }
  ul {
    list-style: none;
  }
  ul > li {
    float: left;
    margin-right: 20px;
    cursor: pointer; //cursor 属性规定要显示的光标的类型（形状）。pointer光标呈现为指示链接的指针（一只手）
  }
  ol {
    display: none;
  }
</style>

<body>
  <ul>
    <li>首页</li>
    <li>新闻</li>
    <li id="item">
      产品
      <ol id="submenu">
        <li>产品A</li>
        <li>产品B</li>
      </ol>
    </li>
  </ul>
  <script>
    // 找元素
    var item = document.getElementById("item");
    var submenu = document.getElementById("submenu");
    // 绑定鼠标移入事件
    item.onmouseover = function () {
      item.style.backgroundColor = "blue";
      submenu.style.display = "block";
    };
    // 绑定鼠标移出事件
    item.onmouseout = function () {
      submenu.style.display = "none";
    };
  </script>
</body>
```

#### 焦点事件

```html
<body>
  <input type="text" name="" id="txt" />
  <span id="msg"></span>
  <script>
    var txt = document.getElementById("txt");
    var msg = document.getElementById("msg");
    // 获得焦点事件
    txt.onfocus = function () {
      console.log("获得焦点");
    };
    // 失去焦点事件
    txt.onblur = function () {
      console.log("失去焦点");
      var val = this.value;
      if (isNaN(val)) {
        msg.innerHTML = "必须是数字";
        msg.style.color = "red";
      }
    };
  </script>
</body>
```

5、 文本框获取焦点，清空信息
文本框失去焦点时，
如果为空 ，提示 不能为空，
如果为非数字，提示不合法，
如果长度小于 6，提示“长度不够”
如果为数字，提示正确

```html
    <input type="text" name="" id="text" value="输入内容（数字）" />
    <script>
        let txt = document.getElementById('text');
        //获取焦点
        txt.onfocus = function() {
                //判断内容是否变化
                if (this.value === this.defaultValue) {
                    this.value = '';
                }
            }
            //失去焦点
        txt.onblur = function() {
            if (this.value === '') {
                alert("内容不能为空");
            } else if (isNaN(this.value)) {
                alert("输入内容不合法")
            } else if (this.value.length < 6) {
                alert("长度不够");
            } else {
                alert("正确");
            }
        }
    </script>
</body>
```
