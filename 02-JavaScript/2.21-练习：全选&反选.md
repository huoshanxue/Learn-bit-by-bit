<!DOCTYPE html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <meta http-equiv="X-UA-Compatible" content="ie=edge" />
    <title>Document</title>
  </head>
  <body>
    <input type="checkbox" name="" id="qx" />
    全选
    <input type="checkbox" name="" id="fx" />
    反选
    <input type="checkbox" name="q" id="" />
    红烧肉
    <input type="checkbox" name="q" id="" />
    西虹市鸡蛋
    <input type="checkbox" name="q" id="" />
    宫爆鸡丁
    <input type="checkbox" name="q" id="" />
    酸菜鱼
    <input type="checkbox" name="q" id="" />
    小炒
    <script>
      var qx = document.querySelector("#qx");
      var fx = document.querySelector("#fx");
      var qs = document.getElementsByName("q");
      //console.dir(qx); qx.checked true/false
      qx.onclick = function () {
        // 获取全选按钮选中状态
        var isChecked = this.checked;
        for (var i = 0; i < qs.length; i++) {
          qs[i].checked = isChecked;
        }
      };

      fx.onclick = function () {
        for (var i = 0; i < qs.length; i++) {
          qs[i].checked = !qs[i].checked;
        }
      };

      // 给各个菜进行单击事件绑定
      var count = 0;
      for (var i = 0; i < qs.length; i++) {
        qs[i].onclick = function () {
          count = 0;
          // 统计所有菜选中个数
          for (var j = 0; j < qs.length; j++) {
            if (qs[j].checked) {
              count++;
            }
          }
          if (count === qs.length) {
            qx.checked = true;
          } else {
            qx.checked = false;
          }
        };
      }
    </script>
  </body>
</html>
