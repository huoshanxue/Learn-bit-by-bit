### 函数默认参数

```javascript
function fn(a = 0, b = 0) {
  return a + b;
}
```

### 剩余运算符

剩余参数 只能为参数的最后一个

```javascript
function fn2(num1, ...num2) {
  //console.log(num2 instanceof Array); //true          instanceof判断是否为某一具体类型
  console.log(num1, num2);
}
fn2(10, 20, 30, 40); // 10 [20,30,40]
```

### 箭头函数

```javascript
//普通函数
function fn(a = 0, b = 0) {
  return a + b;
}
//箭头函数
/*
let arrow = (a = 0,b = 0) => {
     return a + b;
}
*/
// 当函数体只有一条返回语句 大括号和return都可以省略
// 输入参数多于一个 则用()包起来 函数体假如有多条语句需要用 {}
let arrow = (a = 0, b = 0) => a + b;
console.log(arrow(5, 4));
[1, 3, 5].forEach((item) => console.log(item));

// 箭头函数没有this 导致箭头函数this是外层代码块this
let p1 = {
  name: "zizi",
  getName: function () {
    setTimeout(function () {
      console.log(this); //window
    }, 1000);
  },
};
p1.getName();

let p1 = {
  name: "zizi",
  getName: function () {
    setTimeout(() => {
      console.log(this); //p1
    }, 1000);
  },
};
p1.getName();
```
